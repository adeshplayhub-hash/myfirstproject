let memory = { name:"" };

// 🔊 SPEAK
function speak(text){
  let msg = new SpeechSynthesisUtterance(text);
  msg.lang = "en-US";
  speechSynthesis.speak(msg);
}

// 📩 SEND MESSAGE
function send(){
  let input = document.getElementById("input");
  let text = input.value.trim();
  if(text==="") return;

  let chat = document.getElementById("chat");

  // user message
  chat.innerHTML += `
${text}
`;

  input.value = "";

  reply(text.toLowerCase());

  chat.scrollTop = chat.scrollHeight;
}

// 🤖 AI LOGIC
function reply(t){

  let r = "";

  if(t.includes("who made you") || t.includes("creator")){
    r = "🤖 मलाई Adesh Chaudhary ले बनाउनु भएको हो।";
  }

  else if(t.includes("my name is")){
    memory.name = t.split("is")[1].trim();
    r = "Nice to meet you " + memory.name;
  }

  else if(t.includes("what is my name")){
    r = memory.name ? "Your name is " + memory.name : "I don't know your name";
  }

  else if(t.includes("hello")){
    r = "Hello 👋";
  }

  else if(t.includes("namaste")){
    r = "नमस्ते 🙏";
  }

  else if(t.includes("how are you")){
    r = "I am fine 😊";
  }

  else if(t.includes("what is ai")){
    r = "AI means Artificial Intelligence 🤖";
  }

  else{
    r = "🤖 I am learning... '"+t+"'";
  }

  setTimeout(()=>{

    let chat = document.getElementById("chat");

    chat.innerHTML += `
${r}
`;

    speak(r);

    chat.scrollTop = chat.scrollHeight;

  },500);
}

// 🎤 VOICE INPUT
function voice(){

  if(!('webkitSpeechRecognition' in window)){
    alert("Voice not supported");
    return;
  }

  let rec = new webkitSpeechRecognition();
  rec.lang = "en-US";
  rec.start();

  rec.onresult = function(e){
    document.getElementById("input").value =
    e.results[0][0].transcript;
    send();
  };
}
